\chapter{Moduli (reprise)}
\begin{definition}{Bimodulo}
    Siano \(A, B\) anelli. Allora un \({(A,B)}\)\emph{-bimodulo} è un gruppo abeliano
    \({(M, +)}\) con struttura di \(A\)-modulo sinistro e di \(B\)-modulo destro
    tale che \( {(ax)}b = a{(xb)}\) per ogni \(a \in A \), \(b \in B\) e \(x \in
    M\).

    Se \(A=B\) allora si dice \(A\)-bimodulo invece che \({(A,A)}\)-bimodulo.
\end{definition}
\begin{remark}{}
    Un \({(A,B)}\)-bimodulo è anche un \({(B^{op}, A^{op})}\)-bimodulo. In
    particolare
    \[
      A\text{-bimodulo} = A^{op}\text{-bimodulo}
    \]
\end{remark}
\begin{example}{}
    \(A\) come anello è un \(A\)-bimodulo
\end{example}
\begin{example}{}
    Se \(A\) è commutativo, allora se \(M\) è \(A\)-modulo è un \(A\)-bimodulo,
    con la proprietà aggiuntiva che \(ax = xa\) (più in generale se vale questa
    proprietà si dice che il bimodulo è \emph{simmetrico})
\end{example}
\begin{example}{}
    Se \(B\) è una \(A\)-algebra e \(M\) un \(B\)-modulo
    \emph{sinistro}/\emph{destro}, allora \(M\) è un
    \({(B,A)}/{(A,B)}\)-bimodulo.

    Di conseguenza, \(\forall A\) anello e \(\forall M\) \(A\)-modulo
    \emph{sinistro}/\emph{destro}, \(M\) è un
    \({(A,\mathbb{Z})}/{(\mathbb{Z},A)}\)-bimodulo.
\end{example}

\begin{definition}{Omomorfismo di bimoduli}
    Un omomorfismo di \({(A,B)}\)-bimoduli è \(f : M \to N\) omomorfismo sia di
    \(A\)-moduli sinistri che di \(B\)-moduli destri.
\end{definition}

Si ottiene che la categoria \(A\mathtt{-Mod-}B\) degli \({(A,B)}\)-bimoduli è
una categoria abeliana.

\begin{definition}{Prodotto tensoriale}
    Un \emph{prodotto tensoriale} di \(M \in \mathrm{Mod-}A\) e \(N \in
    A\mathtt{-Mod}\) è un gruppo abeliano \(T\) con una funzione \(t: M \times N
    \to T\) che sia \(A\)-bilanciata, cioè \(t\) è \(\mathbb{Z}\)-bilineare e
    \[
      t{(xa, y)} = t{(x, ay)} \quad \forall a \in A,\,\, \forall x \in M,\,\,
      \forall y \in N
    \]
    universale per tale proprietà, ossia \(t\) è
    tale che \(\forall G\) gruppo abeliano e \(\forall f: M \times N \to G\)
    \(A\)-bilanciata \(\exists ! f' : T \to G\) \(\mathbb{Z}\)-lineare tale che
    \(f = f' \circ t\) 
\end{definition}

\begin{proposition}{}
    Siano \({(T, t)}\) e \({(T', t')}\) due prodotti tensoriali di \(M\) e
    \(N\). Allora \(\exists ! i : T \to T'\) \(\mathbb{Z}\)-lineare tale che
    \(t' = i \circ t\) e \(i\) è isomorfismo.

    Inoltre si può prendere \(T := \mathbb{Z}^{{(M \times N)}} / H\). Con
    leggero abuso di notazione in \(\mathbb{Z}^{{(M \times  N)}}\) indico con
    \({(x, y)}\) l'elemento della base indicizzato da \({(x, y)} \in M \times
    N\). Con tale notazione, e \(x, x' \in M\), \(y, y' \in N\) e \(a \in A\) 
    \[
        H = \Span{\underbrace{{(x + x', y)} - {(x,y)} -
        {(x',y)}}_{\text{additività nel primo argomento}}, \underbrace{{(x, y +
y')} - {(x,y)} - {(x, y')}}_{\text{add. nel secondo arg.}}, \underbrace{{(xa,y)}
- {(x, ay)}}_{\text{bilanciataggine}}} 
    \]
    e 
    \[
      t{(x,y)} = {(x,y)} + H
    \]
\end{proposition}
\begin{proof}[Dimostrazione unicità]
    Per la proprietà universale di \(T, t\), \(\exists ! i \) tale che \(t' = i
    \circ t\). Analogamente per \(i' : T' \to T\) e come al solito \(i \circ i'
    = \mathrm{id}_{T'} \) e \(i' \circ i = \mathrm{id}_T\) 
\end{proof}
\begin{proof}{Dimostrazione esistenza}

    ^{{(M \times N)}} \to T\) la proiezione).

    Abbiamo che \(t = \pi \circ j\), con \(j : M \times N \to \mathbb{Z}^{{(M \times
    N)}}\) definita da \({(x, y)} \mapsto {(x,y)}\).

    Finalmente si conclude che
    \[
      f = \tilde{f} \circ j = f' \circ \pi \circ j = f' \circ t
    \]

    allora \({(b b')}z = b{(b' z)}\). Poiché basta verificarlo su un generatore

    Analogamente per \(z {(c c')} = {(z c)}c'\) e \({(bz)}c = b{(zc)}\).
\end{proof}

\begin{remark}{}
    Se \(A\) è commutativo, allora 
    \begin{align*}
        t : M \times N &\to M \otimes_A N \\
        {(x, y)} & \mapsto x \otimes y
    \end{align*}
    è \(A\)-bilineare (oltre che \(A\)-bilanciata ovviamente) perché
    \(t {(ax, y)} = {(ax)} \otimes y = a{(x \otimes y)} = x \otimes {(ay)} =
    t{(x, ay)}\).

    Inoltre, data \(f: M \times N \to P\) \(A\)-bilineare (con \(P \in
    A\mathtt{-Mod}\), allora \(f\) è \(A\)-bilanciata e quindi \(\exists! f' :
    M \otimes_A N \to P\) \(\mathbb{Z}\)-lineare tale che \(f = f' \circ t\)  e
    \(f'\) è \(A\)-lineare perché \(f'{(a{(x \otimes y)})} = f'{({(ax)} \otimes
    y)} = f'{(t{(ax, y)})} = f{(ax, y)} = a f{(x, y)} = \dots = af'{(x \otimes
    y)}\).

    Una conseguenza è che su \(A\) commutativo si può definire \(M \otimes_A N\)
    attaverso la proposizione universale per le funzioni \(A\)-bilineari \(M
    \times N \to P\).
\end{remark}

\begin{definition}{Funtore aggiunto}
    Sia \(F : \mathcal{C} \to \mathcal{D}\) e \(G : \mathcal{D} \to
    \mathcal{D}\) funtori. Si dice che \(F\) è \emph{aggiunto sinistro} di
    \(G\) (o che \(G\) è \emph{aggiunto destro} di \(F\)) e si indica \(F \dashv
    G\) se \(\forall X \in \mathcal{C}\) e \(\forall Y \in \mathcal{D}\)
    \[
      \exists \varphi = \varphi_{X,Y} : \mathcal{D}{(F{(X)}, Y)} \to
      \mathcal{C}{(X, G{(Y)})}
    \]
    biunivoca e naturale in \(X\) e \(Y\), cioè \(\forall f : X' \to X\) in
    \(\mathcal{C}\) e \(\forall g : Y \to Y'\) in \(\mathcal{D}\) il seguente
    diagramma commuta
    \[\begin{tikzcd}
        \mathcal{D}{(F{(X')}, Y)} & \mathcal{D}{(F{(X)}, Y)} &
        \mathcal{D}{(F{(X)}, Y')} \\
        \mathcal{C}{(X', G{(Y)})} & \mathcal{C}{(X, G{(Y)})} & \mathcal{C}{(X,
        G{(Y')})}
        \arrow["{\varphi_{X', Y}}"', from=1-1, to=2-1]
        \arrow["{F(f)^*}"', from=1-2, to=1-1]
        \arrow["{g_*}", from=1-2, to=1-3]
        \arrow["{\varphi_{X,Y}}"', from=1-2, to=2-2]
        \arrow["{\varphi_{X, Y'}}", from=1-3, to=2-3]
        \arrow["{f^*}"', from=2-2, to=2-1]
        \arrow["{G(g)_*}", from=2-2, to=2-3]
    \end{tikzcd}\]
    ossia \(\forall h \in \mathcal{D}{(F{(X)}, Y)}\), \[\varphi_{X, Y'}  {(g \circ h)} =
    G{(g)} \circ \varphi_{X, Y}  {(h)} \text{ e } \varphi _{X',Y} {(h \circ F{(f)})}
    = \varphi _{X, Y} {(h)}
    \circ f\] 
\end{definition}

\begin{remark}{}
    Una \(\varphi \) come sopra definisce una trasformazione naturale
    (isomorfismo) \(\mathcal{D}{(F{(-)}, =)} \Rightarrow \mathcal{C}{(-, G{(=)})}\) tra
    funtori \(\mathcal{C}^{op} \times \mathcal{D} \to \mathtt{Set}\) (o
    viceversa dato un tale isomorfismo naturale si ottiene un'aggiunzione tra
    \(F\) e \(G\)). Questo equivale a dire che, per ogni \( X' \overset{f}{\to }
    X\) e \(Y \overset{g}{\to } Y'\),  il seguente diagramma commuta
    \[\begin{tikzcd}
	\mathcal{D}{(F{(X)}, Y)} & \mathcal{C}{(X, G{(Y)})} \\
	\mathcal{D}{(F{(X')}, Y')} & \mathcal{C}{(X', G{(Y')})}
	\arrow["{\varphi_{X,Y}}", from=1-1, to=1-2]
	\arrow["{g_* \circ F(f)^*}"', from=1-1, to=2-1]
	\arrow["{G(g)_*\circ f^*}", from=1-2, to=2-2]
	\arrow["{\varphi_{X', Y'}}"', from=2-1, to=2-2]
\end{tikzcd}\]
    ossia che \(\forall h \in \mathcal{D}{(F{(X)}, Y)}\), si ha che
    \(\varphi_{X', Y'} \circ g_* \circ F{(f)}^{*} {(h)} = G{(g)}_* \circ f^{*}
    \circ \varphi_{X, Y} {(h)}\), ossia \(\varphi_{X',Y'} {(g \circ h \circ
    F{(f)})} = G{(g)} \circ \varphi_{X, Y} {(h)} \circ f\) 
\end{remark}

\begin{remark}{}
    \(F \dashv G \iff G^{op} \dashv F^{op}\) 
\end{remark}

\begin{example}{}

    \to \mathcal{C}\) è un quasi-inverso di \(F\), allora \(F \dashv G \) e \(G
    \dashv F\)

        \mathcal{D}\), \[\mathcal{D}{(F{(X)}, Y)}
            \underset{\sim}{\overset{G}{\to}} \mathcal{C}{(G{(F{(X)})},
    G{(Y)})}\underset{\sim}{\overset{\alpha_X^{*}}{\to}} \mathcal{C}{(X, G{(Y)})}
\] 
    se \(\alpha: \mathrm{id}_\mathcal{C} \to G \circ F\) isomorfismo (\(\implies
    \alpha_X :X \to G \circ F{(X)}\) isomorfismo in \(\mathcal{C}\))
    \end{proof}
\end{example}

\begin{proposition}{}
    Se \(F \overset{\varphi }{\dashv } G\) e \(F, F' : \mathcal{C} \to
    \mathcal{D}\), allora \(F' \dashv G \iff F' \cong F\) 
\end{proposition}
\begin{proof}\( \)
\begin{itemize}
    \item[\(\implies \)] Cerco \(\alpha_X \in \mathcal{D}{(F{(X)}, F'{(X)})}\) e
        possiamo prendere
        \[
          \alpha_X = \varphi_{X,F'{(X)}} ^{-1} {\left( \varphi '_{X, F'{(X)}}
          {(1_{F'{(X)}} )} \right)} 
        \]
    \item[\(\impliedby \)] Sia \(\alpha : F \to F'\) l'isomorfismo. Allora
        voglio definire \(\varphi'_{X, Y}  : \mathcal{D}{(F'{(X)} \to Y)} \to
        \mathcal{C}{(X, G{(Y)})}\) per ogni \(X \in \mathcal{C}\) e \(Y \in
        \mathcal{D}\) e possiamo definirlo come \(\varphi'_{X, Y} = \varphi_{X,
        Y} \circ \alpha_X^{*}\)
\end{itemize}

\begin{theorem}{}
    Siano \(\mathcal{C}\) e \(\mathcal{D}\) categorie qualunque e siano \(F, G\)
    funtori tali che
    \[
      {(F : \mathcal{C} \to \mathcal{D})} \dashv {(G : \mathcal{D} \to
      \mathcal{C})}
    \]
    Allora \(F\) preserva i colimiti e \(G\) preserva i limiti.
\end{theorem}

\begin{proof}[idea della dimostrazione]
    Sia \(I : \mathcal{L} \to \mathcal{D}\) un funtore (con \(\mathcal{L}\)
    piccola) con \( \lim I = {(Z \in \mathcal{D}, \alpha: K_Z \to I)}\). Allora
    per definizione di limite si ha che, \(\forall Y \in \mathcal{D}\),
    \begin{align*}
        \delta_{Y, Z} : \mathcal{D}{(Y, Z)} &\longrightarrow
        \mathtt{Fun}{(\mathcal{L}, \mathcal{D})}{(K_Y, I)} \\
        g &\longmapsto \delta_{Y, Z} (g) = \alpha \circ K_g
    \end{align*}
    è biunivoca. Vogliamo dimostrare che \( \lim {(G \circ I)} = {(G{(Z)}, G
    \circ \alpha : K_{G{(Z)}} \to G \circ I)}\), ossia che \(\forall X \in
    \mathcal{C}\), 
    \begin{align*}
        \gamma_{X, Z} : \mathcal{C}{(X, G{(Z)})} &\longrightarrow
        \mathtt{Fun}{(\mathcal{L}, \mathcal{C})}{(K_X, G \circ I)} \\
        f &\longmapsto \gamma_{X, Z} (f) = g \circ \alpha \circ K_f
    \end{align*}
    è biunivoca. A tale scopo notiamo che abbiamo il seguente diagramma
    commutativo:
    \[\begin{tikzcd}
	\mathcal{C}{(X, G{(Z)})} & \mathtt{Fun}{(\mathcal{L}, \mathcal{C})}{(K_X, G
    \circ I)} \\
	\mathcal{D}{(F{(X)}, Z)} & \mathtt{Fun}{(\mathcal{L},
    \mathcal{D})}{(K_{F{(X)}} , I)}
	\arrow["{\gamma_{X, Z}}", from=1-1, to=1-2]
	\arrow["{\varphi_{X,Z}}"', tail reversed, from=1-1, to=2-1]
	\arrow["{\delta_{F(X),Z}}"', tail reversed, from=2-1, to=2-2]
	\arrow["{\tilde{\varphi}}"', from=2-2, to=1-2]
\end{tikzcd} \]
    dove \(\tilde{\varphi}\) è definita nel seguente modo per ogni \(\beta \in
    \mathtt{Fun}{(\mathcal{L}, \mathcal{D})}{(K_{F{(X)}} , I)}\), ossia per ogni
    trasformazione naturale \(\beta : K_{F{(X)}} \to I \) 
    \[
      \forall L \in \mathcal{L} \quad \tilde{\varphi}{(\beta)}_L := \varphi_{X,
      I{(L)}} {(\beta_L)} : X \to G{(I{(L)})}
    \]
    si verifica che \(\tilde{\varphi}\) è ben definita ed è biunivoca,
    il diagramma commuta e dunque \(\gamma_{X, Z}\) è biunivoca.
\end{proof}

\end{corollary}

\begin{proposition}{}

    B\mathtt{-Mod}\). Allora 
    \[
      {(M \otimes_A N)}\otimes_B P \cong M \otimes_A {(N \otimes_B P)}
    \]

    \otimes z)}\) 
\end{proposition}

\begin{proof}{}
    Fissato \(z \in P\), sia 
    \begin{align*}
        f_z: M \times N &\longrightarrow M \otimes_A {(N \otimes_B P)} \\
        {(x,y)} &\longmapsto f_z({(x,y)}) = x \otimes {(y \otimes z)}
    \end{align*}
    allora \(f_z\) è \(A\)-bilanciata e dunque per la proprietà universale del prodotto
    tensoriale \(\exists ! f_z' : M \otimes_A N \to M \otimes_A {(N \otimes_B
    P)}\) \(\mathbb{Z}\)-lineare tale che \(x \otimes y \mapsto x \otimes {(y
    \otimes z)}\). Sia ora 
    \begin{align*}
        f: {(M \otimes_A N)} \times P &\longrightarrow M \otimes_A {(N \otimes_B
        P)} \\
        {(w,z)} &\longmapsto f({(w,z)}) = f'_z{(w)}
    \end{align*}
    Allora \(f\) è \(B\)-bilanciata. Infatti se \(w = x \otimes y\) (basta
    verificare in questo caso) 
    \begin{align*}
        f{(wb, z)}&= f'_z{(wb)} = f'_z {({(x \otimes y)}b)} = f'_z {(x \otimes yb)}

      \otimes_B P)} \) \(\mathbb{Z}\)-bilineare t.c. \(f'{({(x \otimes y)}
      \otimes z)} = x \otimes {(y \otimes z)}
    \)
\end{proof}

\section{Funtori indotti da omomorfismi di anelli}

A questo punto, dato un omomorfismo di anelli \(f : A \to B\), esso induce i
funtori
\begin{itemize}[label = --]
    \item \(f^{*}: B\mathtt{-Mod} \to A\mathtt{-Mod}\) (restrizione degli
        scalari)
    \item \(f_! := B \otimes_A - : A\mathtt{-Mod} \to B\mathtt{-Mod}\)
        (estensione degli scalari, considerando \(B \in B\mathtt{-Mod-}A\))
    \item \(f_* := \mathrm{Hom}_A {(B, -)} : A\mathtt{-Mod} \to B\mathtt{-Mod}\)
        (coestensione degli scalari, considerando \(B \in A\mathtt{-Mod-}B\))
\end{itemize}

Come notazione si può usare \(f_*\) invece di \(f^{*}\), \(f^{*}\) invece di
\(f_!\) e \(f^{!}\) invece di \(f_*\) 

\begin{proposition}{}
    Sia \(f: A \to B\) come sopra, ossia \(f \in \mathtt{Rng}{(A,
    B)}\), allora
    \[
      f_! \dashv f^{*} \dashv f_*
    \]
\end{proposition}

\begin{proof}{}
    \(\forall N \in A\mathtt{-Mod}\) e \(\forall B \in B\mathtt{-Mod}\) devo
    dimostrare che
\begin{enumerate}[label = \arabic*.]
    \item \(\mathrm{Hom}_B{(f_!{(N)}, P)} \cong \mathrm{Hom}_A {(N, f^{*}{(P)})}\) 
    \item \(\mathrm{Hom}_A {(f^{*}{(P)}, N)} \cong \mathrm{Hom}_B{(P, f_*{(N)})}\) 
\end{enumerate}

\begin{proof}[1.]
    \(\mathrm{Hom}_B{(f_!{(N)}, P)} = \mathrm{Hom}_B{(B\otimes_A N, P)} \cong
    \mathrm{Hom}_A {(N, \mathrm{Hom}_B{(B, P)})} \) per il
    corollario~\ref{cor:mod_tens} ma ora \(\mathrm{Hom_B{(B, P)}} \cong P\) come
    \(A\)-modulo, dunque abbiamo che il precedente 
    \[
      \mathrm{Hom}_B{(f_!{(N)}, P)} \cong \mathrm{Hom}_A {(N, P)} =
      \mathrm{Hom}_A {(N, f^{*}{(P)})}
    \]
    dove nell'ultimo passaggio si è usato che effettivamente
    \(\mathrm{Hom}_A{(N, P)}\) è la restrizione degli scalari.
\end{proof}

\begin{proof}[2.]
    \begin{align*}
        \mathrm{Hom}_A {(f^{*}{(P)}, N)} &\overset{\text{\(P\) \(A\)-modulo}}{=}
        \mathrm{Hom}_A {(P, N)} \cong \mathrm{Hom}_A {(B \otimes_B P, N)} \cong \\ &\cong
      \mathrm{Hom}_B{(P, \mathrm{Hom}_A{(B, N)})} = \mathrm{Hom}_B{(P, f_*{(N)})}
    \end{align*}
\end{proof}
\end{proof}

\begin{definition}{Proiettività e iniettività}
    Sia \(\mathcal{A}\) una categoria abeliana e \(X \in \mathcal{A}\). Allora
    \(X\) si dice \emph{proiettivo} se \(\mathcal{A}{(X, -)} : \mathcal{A} \to
    \mathtt{Ab}\) è esatto (ossia equivalentemente se \(\mathcal{A}{(X, -)}\)
    preserva gli epi, cioè \(\forall f : Y \twoheadrightarrow Z\) epimorfismi di
    \(\mathcal{A}\), allora \(f_* : \mathcal{A}{(X, Z)} \to \mathcal{A}{(X,
    Z)}\) è suriettivo, ossia per ogni \(g : X \to Z\) esiste \(s : X \to Y\)
    tale che \(f \circ s = g\))

    \(X\) si dice \emph{iniettivo} (in \(\mathcal{A}\)) se è proiettivo in
    \(\mathcal{A}^{op}\), cioè se \(\mathcal{A}{(-, X)} : \mathcal{A}^{op} \to
    \mathtt{Ab}\) è esatto. (Equivalentemente \(\mathcal{A}{(-, X)}\) preserva
    gli epimorfismi, ossia \(\forall f : Y \rightarrowtail Z\) monomorfismo di
    \(\mathcal{A}\), \(f^{*} : \mathcal{A}{(Z, X)} \to \mathcal{A}{(Y, X)}\) è
    suriettivo).
\end{definition}

\begin{remark}{}
    Se \(0 \to X \overset{f}{\to} Y \overset{g}{\to} Z \to 0\) è esatta, \(Z\) è
    proiettivo oppure \(X\) è iniettivo, allora la successione si spezza. Ad
    esempio se \(Z\) è proiettivo, allora \(g : Y \twoheadrightarrow Z \) e
    esiste \(\mathrm{id}_Z : Z \to Z\), allora \(\exists s : Z \to Y\) tale che
    \(g \circ s = 1_Z\).

    Quindi otteniamo che \(\mathcal{A}\) è semisemplice se e solo se tutti gli
    oggetti di \(\mathcal{A}\) sono proiettivi se e solo se tutti gli oggetti di
    \(\mathcal{A}\) sono iniettivi.
\end{remark}

\begin{definition}{Moduli piatti}
    \(M \in \mathtt{Mod}-A\) è \emph{piatto} (su \(\mathcal{A}\)) se 
    \(M \otimes_A - : A\mathtt{-Mod} \to \mathtt{Ab}\) è esatto.
    Equivalentemente \(M \otimes_A -\) preserva i monomorfismi, cioè se \(f : N
    \rightarrowtail P\) iniettivo in \(A\mathtt{-Mod}\), allora \(f_* :
    M\otimes_A N \to M \otimes_A P\) è iniettivo in \(\mathtt{Ab}\).
\end{definition}

\begin{proposition}{} \(\) 
\begin{enumerate}[label = \arabic*.]
    \item Sia \(\mathcal{A}\) una categoria abeliana e \(X_\lambda \in
        \mathcal{A}\) per ogni \(\lambda \in \Lambda\). Supponiamo inoltre che
        esista \(X = \coprod_{\lambda \in \Lambda} X_\lambda\) (resp. \(X = \prod_{\lambda
        \in \Lambda} X_\lambda\)). Allora \(X\) è proiettivo (resp. iniettivo) se
        e solo se tutti gli \(X_\lambda\) sono proiettivi (resp. iniettivi).
    \item Sia \(A\) un anello, \(M_\lambda \in \mathtt{Mod}-A \) per ogni
        \(\lambda \in \Lambda\). Allora 
        \[
          \bigoplus_{\lambda \in \Lambda} M_\lambda \text{ è piatto } \iff
          M_\lambda \text{ è piatto } \forall  \lambda \in \Lambda
        \]
\end{enumerate}
\end{proposition}
\begin{proof}{} \( \) 
\begin{enumerate}[label = \arabic*.]
    \item Basta dimostrare il caso \(X = \coprod_{\lambda \in \Lambda}
        X_\lambda\). In tale caso sia \(f : Y \twoheadrightarrow Z\) un
        epimorfismo in \(\mathcal{A}\). Allora \(X\) è proiettivo se e solo se
        \(f_* : \mathcal{A}{(X, Y)} \to \mathcal{A}{(X, Z)}\) è suriettivo. Ma
        \[
          \mathcal{A}{(X, Y)} = \mathcal{A}{\left( \prod_{\lambda \in \Lambda}
          X_\lambda, Y \right)} \overset{\text{ prop. univ. \(\coprod\)}}{\cong}
          \prod_{\lambda \in \Lambda} \mathcal{A}{(X_\lambda, Y)}
        \]
        quindi ora poiché il seguente diagramma commuta, 

\[\begin{tikzcd}
	\mathcal{A}{(X, Y)} & \mathcal{A}{(X, Z)} \\
	\prod_{\lambda \in \Lambda} \mathcal{A}{(X_\lambda, Y)} & \prod_{\lambda \in \Lambda} \mathcal{A}{(X_\lambda, Z)}
	\arrow["{f_*}", from=1-1, to=1-2]
	\arrow["\cong"{marking, allow upside down}, draw=none, from=1-1, to=2-1]
	\arrow["\cong"{marking, allow upside down}, draw=none, from=1-2, to=2-2]
	\arrow["{\prod_{\lambda \in \Lambda}f_*}", from=2-1, to=2-2]
	\arrow["{(g_\lambda)_{\lambda \in \Lambda} \mapsto (f \circ g_\lambda)_{\lambda \in \Lambda}}"', shift right=2, draw=none, from=2-1, to=2-2]
\end{tikzcd}\]

        La mappa \(f_*\) è suriettiva se e solo se \(\prod_{\lambda \in \Lambda}
        f_*\) è suriettiva. Ma quest'ultima lo è se tutte le sue componenti lo
        sono, ossia se ogni \(X_\lambda\) è proiettivo.
    \item \(M\) è piatto se e solo se \(\forall f : N \to P\) iniettivo in
        \(A\mathtt{-Mod}\), \(f_* : M \otimes_A N \to M \otimes_A P\) è
        iniettivo in \(\mathtt{Ab}\). Poiché \(- \otimes_A N\) preserva i
        coprodotti, \(M \otimes_A N \cong \bigoplus_{\lambda \in \Lambda} M_\lambda
        \otimes N\). Allora
\[\begin{tikzcd}
	M \otimes_A N & M \otimes_A P \\
	\bigoplus_{\lambda \in \Lambda} M_{\lambda} \otimes_A N & \bigoplus_{\lambda \in \Lambda} M_\lambda \otimes_A P
	\arrow["{f_*}", from=1-1, to=1-2]
	\arrow["\cong"{marking, allow upside down}, draw=none, from=1-1, to=2-1]
	\arrow["\cong"{marking, allow upside down}, draw=none, from=1-2, to=2-2]
	\arrow["{\bigoplus_{\lambda \in \Lambda}f_*}", from=2-1, to=2-2]
\end{tikzcd}\]

    ma ora \(f_*\) è iniettivo \(\iff\) se \(\bigoplus_{\lambda \in \Lambda}
    f_*\) è iniettivo \(\iff\) ogni componente di \(\bigoplus_{\lambda \in
    \Lambda} f_* \) è iniettiva \(\iff\) ogni \(M_\lambda\) è piatto.
\end{enumerate}
\end{proof}

\begin{corollary}{} \(\) 
\begin{enumerate}[label = \arabic*.]
    \item Un \(A\)-modulo è proiettivo se e solo se è addendo diretto di un
        modulo libero.
    \item Ogni modulo proiettivo è piatto.
\end{enumerate}
\end{corollary}
\begin{proof}{}
    \(A \in A\mathtt{-Mod}\) è proiettivo e piatto perché \(\mathrm{Hom}_A {(A,
    -)} : A\mathtt{-Mod} \to \mathtt{Ab}\) è isomorfo al funtore dimenticante, e
    dunque è esatto. Analogamente \(- \otimes_A A : \mathtt{Mod-}A \to
    \mathtt{Ab}\) è isomorfo al funtore dimenticante e dunque è esatto.

    Sia \(\mathcal{P}\) una collezione di oggetti (chiusa per isomorfismi) di
    \(A\mathtt{-Mod}\) tale che \(A \in \mathcal{P}\) e tali che se \(M_\lambda
    \in A\mathtt{-Mod}\) (con \(\lambda \in \Lambda\)), si ha che 
    \[
      \bigoplus_{\lambda \in \Lambda} M_\lambda \in \mathcal{P} \iff M_\lambda
      \in \mathcal{P} \quad \forall \lambda \in \Lambda
    \]
    Allora ogni \(M\) addendo diretto di un modulo libero è tale che \(M \in
    \mathcal{P}\). Infatti in tal caso \(\exists N \in A\mathtt{-Mod}\) tale che
    \(M \oplus N\) è libero, dunque \(M \oplus N \cong A^{{(\Lambda)}}\) per
    qualche \(\Lambda\), e dunque \(A \in \mathcal{P} \implies A^{{(\Lambda)}}
    \in \mathcal{P} \implies M \oplus N \in \mathcal{P} \implies  M \in
    \mathcal{P}\), quindi sia i proiettivi che i piatti contengono gli addendi
    diretti dei moduli liberi.

    Per concludere basta dimostrare che se \(M \in A\mathtt{-Mod}\) è
    proiettivo, allora è addendo diretto di un modulo libero. Questo è vero
    perché \(\exists L \in A\mathtt{-Mod}\) libero, e \(N \subseteq L \)
    sottomodulo tale che \(M \cong L/M\), cioè esiste una successione esatta
    \[
      0 \to N \to L \to M \to 0
    \]
    ma poiché \(M\) è proiettivo la successione si spezza, e dunque \(L \cong M
    \oplus N\) 
\end{proof}

\begin{example}{}
    Sia \(A = \mathbb{Z} / 6 \mathbb{Z}\) (semisemplice), allora
    \(\Span{\overline{2}} \) è un \(A\)-modulo proiettivo non libero.
\end{example}

\begin{example}{}
    Sia \(A\) un dominio d'integrità che non sia un campo, sia \(0 \neq a \in A
    \sminus A^{*}\). Allora \(A / (a)\) come \(A\)-modulo non è piatto, infatti
    \(A \to A ; b \mapsto ab\) è omomorfismo iniettivo in \(A\mathtt{-Mod}\)
    tale che 
    \[ \begin{tikzcd}
	A \otimes_A A / (a) & A \otimes-A A / (a) \\
	A / (a) & A / (a)
	\arrow["{\varphi_{a*}}", from=1-1, to=1-2]
	\arrow["\cong"{marking, allow upside down}, draw=none, from=1-1, to=2-1]
	\arrow["\cong"{marking, allow upside down}, draw=none, from=1-2, to=2-2]
	\arrow["0", from=2-1, to=2-2]
\end{tikzcd} \]
    dove chiaramente \(0 : x \mapsto ax\) in \(A / (a)\) non è iniettivo.
\end{example}

\begin{proposition}{}
    \(M \in A\mathtt{-Mod}\) è iniettivo se solo se per ogni \(I \subseteq A \)
    ideale sinistro e \(\forall g : I \to M\) \(A\)-lineare, \(\exists x \in M\)
    tale che \(g{(a)} = ax\) per ogni \(a \in I\).
\end{proposition}
\begin{proof}\( \)
\begin{itemize}
    \item[\(\implies \)] Poiché \(M\) è iniettivo, \(\begin{tikzcd}
	I & A \\
	M
	\arrow[tail, from=1-1, to=1-2]
	\arrow["g"', from=1-1, to=2-1]
	\arrow["{g'}", dashed, from=1-2, to=2-1]
\end{tikzcd}\) commuta, ossia esiste \(g': A \to M\) \\\(A\)-lineare tale che
\(g'|_I = g\). Allora \(x := g'{(1)}\) e abbiamo che \(ax = g'{(a)} =g{(a)}\)
per ogni \(a \in I\).
    \item[\(\impliedby \)] Dato \(N_{0} \subseteq N \) sottomodulo e \(f_{0} :
        N_{0} \to M\) \(A\)-lineare, basta dimostrare che \(\exists f : N \to
        M\) \(A\)-lineare tale che \(f|_{N_{0}}  = f_{0}\).

        Sia ora \begin{align*}U := \{{(N', f')} | N_{0} \subseteq N' \subseteq N \text{
        sottomodulo e } f' : N' \to M \\\text{ è \(A\)-lineare ed è t.c. }
f'|_{N_{0}} =f_{0}\}\end{align*}
        Dove \(U\) è ordinato da \({(N', f')} \le {(N'', f'')} \iff N' \subseteq
        N''\) e \(f''|_{N'} = f'\).
        Inoltre \({(N_{0}, f_{0})} \in U\) che dunque non è vuoto. Inoltre ogni
        catena \({(N_\lambda, f_\lambda)}\) in \(U\) (per \(\lambda \in
        \Lambda\)) ha un maggiorante \(N' = \bigcup_{\lambda \in \Lambda}
        N_{\lambda}, f'\) tale che \(f'|_{N_{\lambda} } = f_{\lambda} \). % TODO eser forse
        
        Allora ci sono le ipotesi del lemma di Zorn, per cui esiste un elemento
        massimale \({(N_{1}, f_{1})}\in U\). Ora dobbiamo solo dimostrare che
        \(N_{1} = N\). Supponiamo per assurdo che \(N_{1} \subsetneq N\). Allora
        \(\exists y \in N \sminus N_{1}\). Sia ora \(I = \{a \in A : ay \in
        N_{1}\} \subseteq A \) ideale sinistro. \(g : I \to M\) data da \(a
        \mapsto f_{1}{(ay)}\) è \(A\)-lineare e per ipotesi \(\exists x \in M\)
        tale che \(f_{1}{(ay)} = g{(a)} = ax\) per ogni \(a \in I\). Allora
        possiamo definire \(N_{2} := N_{1} + Ay\) e \(f_{2} : N_{2} \to M\) data
        da \(y_{1} + ay \mapsto f_{1}{(y_{1})} + ax\) che è ben definita
        (lasciato per esercizio), \(A\)-lineare, e \({(N_{1}, f_{1})} \lneqq
        {(N_{2}, f_{2})}\), che è assurdo.
        
\end{itemize}
\end{proof}

\subsection{Moduli divisibili e senza torsione}

\begin{definition}{Modulo divisibile}
    Sia \(A\) un dominio d'integrità, \(M \in A\mathtt{-Mod}\). Allora \(M\) è
    \emph{divisibile} se \(\forall x \in M\) e \(\forall 0 \neq a \in A\),
    allora \(\exists y \in M\) tale che \(x = ay\) 
\end{definition}

\begin{remark}{}
    Un modulo \(M\) è divisibile se e solo se \(\forall 0 \neq a \in A\) la
    funzione \(M \to M\) definita da \(x \mapsto ax\) è suriettiva. Analogamente
    \(M\) è senza torsione se e solo se tale funzione è iniettiva.
\end{remark}

\begin{remark}{}
    Se \(M\) è divisibile e \(M' \subseteq M \) sottomodulo, allora \(M / M'\) è
    ancora divisibile.
\end{remark}

\begin{example}{}
    Il campo dei quozienti \(Q{(A)}\) è divisibile in \(A\mathtt{-Mod}\). Di
    conseguenza anche \(Q{(A)} / A\) è divisibile.
\end{example}

\begin{proposition}{}/\label{prop:M-inj-div}
    Sia \(A\) un dominio d'integrità e \(M \in A\mathtt{-Mod}\). Allora se \(M\) è
    iniettivo, allora \(M\) è divisibile. Inoltre se \(A\) è PID vale anche
    l'altra implicazione.
\end{proposition}
\begin{proof}{}
    \(M\) è iniettivo se e solo se \(\forall I \subseteq A \) ideale e \(\forall
    g : I \to M\) \(A\)-lineare, allora \(\exists x \in M\) tale che \(g {(a)} =
    ax\) per ogni \(a \in I\).

    \(M\) è d'altra parte divisibile se e solo se \(\forall 0 \neq I \subseteq A
    \) ideale \textbf{principale} e \(\forall g : I \to M\) \(A\)-lineare,
    allora \(\exists x \in M\) tale che \(g{(a)} = ax\) per ogni \(a \in I\).
    Questa definizione equivalente vale perché in tal modo \(I = {(b)}\), con
    \(0 \neq b \in A\), allora \(I \cong A\) in \(A\mathtt{-Mod}\), quindi \(g :
    I \to M\) \(A\)-lineare è determinata da \(g{(b)} = y \in M\). Allora
    chiedere l'esistenza dell'\(x\) di cui sopra equivale a dire che \(\exists x
    \in M\) tale che \(g{(ab)} = abx\) per ogni \(ab \in I\), ossia \(y = bx\).

    La proposizione segue ovviamente dalla similarità delle due
    caratterizzazioni.
\end{proof}

\begin{definition}{Avere abbastanza iniettivi o proiettivi}
    Una categoria abeliana \(\mathcal{A}\) \emph{ha abbastanza iniettivi} se
    \(\forall X \in \mathcal{A}\) \(\exists X \rightarrowtail Y\) mono con \(Y\)
    iniettivo.

    Dualmente \(\mathcal{A}\) \emph{ha abbastanza proiettivi} se
    \(\mathcal{A}^{op}\) ha abbastanza iniettivi, cioè se \(\forall X \in
    \mathcal{A}\) \(\exists Y \twoheadrightarrow X\) epi (di \(\mathcal{A}\))
    con \(Y\) proiettivo (in \(\mathcal{A}\)).
\end{definition}

\begin{example}{}
    \(A\mathtt{-Mod}\) ha abbastanza proiettivi perché ogni modulo è quoziente
    di un modulo libero, e i moduli liberi sono proiettivi.
\end{example}

\begin{proposition}{}
    Sia \(f : A \to B\) un omomorfismo di anelli. Se \(A\mathtt{-Mod}\) ha
    abbastanza iniettivi, allora \(B\mathtt{-Mod}\) ha abbastanza iniettivi.
\end{proposition}

\begin{proof}{}
    Sia \(N \in B\mathtt{-Mod}\). Per ipotesi \(\exists i : f^{*}{(N)}
    \twoheadrightarrow M\) mono in \(A\mathtt{-Mod}\) con \(M\) iniettivo.
    Sappiamo che \(f^{*} \dashv f_* = \mathrm{Hom}_A{(B, -)}\). Allora \(f_*\) è
    esatto a sinistra e dunque \(f_*{(i)}\) è mono. Sia \(\eta :
    \mathrm{id}_{B\mathtt{-Mod}} \to f_* \circ f^{*}\) unità dell'aggiunzione.
    Allora \(\eta_N : N \to f_* {(f^{*}{(N)})} = \mathrm{Hom}_A{(B, N)}\). Viene
    lasciato in esercizio mostrare che effettivamente tale mappa è \(y \mapsto
    {(b \mapsto by)}\). 

    Se \(y \in \mathrm{Ker}{(\eta_N)}\) allora \(\eta_N{(y)}{(1)} = 0\), quindi
    \(\eta_N\) è iniettiva, allora
    \[
      N \overset{\eta_N}{\rightarrowtail } f_*{(f^{*}{(N)})}
      \overset{f_*{(i)}}{\rightarrowtail } f_*{(M)}
    \]
    dunque basta dimostrare che \(f_*{(M)}\) è iniettivo, cioè \\
    \(\mathrm{Hom}_B{(-, f_*{(M)})} \cong \mathrm{Hom}_A{(f^{*}{(-)}, M)}\) è esatto. Questo è vero poiché sono esatti
    sia \(f^{*}\) che \(\mathrm{Hom}_A{(-, M)}\).
\end{proof}

\begin{proposition}{}
    \(\mathtt{Ab} \cong \mathbb{Z}\mathtt{-Mod}\) ha abbastanza iniettivi.
\end{proposition}
\begin{corollary}{}
    Per ogni anello \(A\), \(A\mathtt{-Mod}\) ha abbastanza iniettivi.
\end{corollary}
\begin{proof}{}
    Per la proposizione~\ref{prop:M-inj-div}, \(G\) è iniettivo se e solo se è
    divisibile (perché \(\mathbb{Z}\) è PID). So che \(\mathbb{Q}/\mathbb{Z} \in
    \mathtt{Ab}\) è divisibile (e allora anche \(\mathbb{Q}/\mathbb{Z} ^{\Lambda}
    \) lo è per ogni \(\Lambda\)) allora per ogni \(0 \neq a \in G\), voglio
    trovare \(f_a : G
    \to \mathbb{Q}/\mathbb{Z}\) è omomorfismo tale che \(f_a{(a)} \neq 0\). A
    Questo punto si può definire 
    \begin{align*}
        f: G &\longrightarrow {(\mathbb{Q}/\mathbb{Z})}^{G \sminus \{0\} } \\
        b &\longmapsto f(b) = {(a \mapsto f_a{(b)})}
    \end{align*}
    omomorfismo iniettivo. Per trovare tale \(f_a\) considero \(\Span{a}
    \hookrightarrow G\). Esiste \(g_a : \Span{a} \to \mathbb{Q}/\mathbb{Z}\)
    omomorfismo tale che \(g{(a)}{(a)} \neq 0\). Allora poiché
    \(\mathbb{Q}/\mathbb{Z}\) è iniettivo esiste \(f_a : G \to
    \mathbb{Q}/\mathbb{Z}\) tale che \(f_a|_{\Span{a} } = g_a\) 
\end{proof}

\begin{remark}{}
    \(\mathrm{Hom}{(-, \mathbb{Q}/\mathbb{Z})} : \mathtt{Ab}^{op} \to
    \mathtt{Ab}\) è esatto (poiché \(\mathbb{Q}/\mathbb{Z}\) è iniettivo) e
    fedele.
    
    Infatti, dato \(f : G \to G'\) in \(\mathtt{Ab}\) tale che \(\widehat{f} = f^{*}
    : \widehat{G}' := \mathrm{Hom}{(G', \mathbb{Q}/\mathbb{Z})} \to \mathrm{Hom}{(G,
    \mathbb{Q}/\mathbb{Z})} = G\), allora se \(\widehat{f} = 0\), segue che \(f =
    0\). Infatti, \(\forall h \in \widehat{G}'\), \(f^{*}{(h)} = h \circ f = 0\).
    Allora se \(\exists a \in G : f{(a)} = a' \neq 0\) allora posso prendere una
    \(h \in \widehat{G}'\) (la \(f_{a'}\) della dimostrazione sopra) tale che
    \(h{(a')} = h{(f{(a)})} \neq 0\). Allora necessariamente \(f = 0\).
\end{remark}

\begin{proposition}{}
    Le seguenti condizioni sono equivalenti
\begin{enumerate}[label = \roman*)]
    \item \(M \in \mathtt{Mod}-A\) è piatto
    \item \(\widehat{M} \in A\mathtt{-Mod}\) è iniettivo
    \item \(\forall I \subseteq A \) ideale sinistro, \(M \otimes_A I \to M\)
        dato da \(x \otimes a \mapsto xa\) è iniettivo.
\end{enumerate}
\end{proposition}

\begin{proof}{}
    \(M\) è piatto se e solo se \(\forall f : N \rightarrowtail  P\) mono in
    \(A\mathtt{-Mod}\), \(f_* : M\otimes_A N \to M\otimes_A P\) è
    iniettivo in \(\mathtt{Ab}\). Questo è equivalente a dire che 
    \(\widehat{f}_* : \widehat{M \otimes_A P} \to \widehat{M \otimes_A N}\) è
    suriettivo, dove \(\widehat{M \otimes_A P} = \mathrm{Hom}{(M \otimes_A P,
    \mathbb{Q}/\mathbb{Z})} \cong \mathrm{Hom}_A{(P, \widehat{M})}\) e \(\widehat{M
\otimes_A N} \cong \mathrm{Hom}_A{(N, \widehat{M})}\). Ma questo è equivalente a
dire che \(\hat{m}\) è iniettivo in \(A\mathtt{-Mod}\).

    Per \(iii)\)  basta considerare \(f : I \to A\) inclusione, con \(I\) ideale
    sinistro.
\end{proof}
